async function loadData(){

const res = await fetch("/api/admin/summary")
const data = await res.json()

document.getElementById("online").innerText = data.onlineUsers
document.getElementById("waiting").innerText = data.waitingUsers
document.getElementById("countries").innerText = data.countriesConnected

const tickets = document.getElementById("tickets")
tickets.innerHTML = JSON.stringify(data.tickets, null, 2)

const reports = document.getElementById("reports")
reports.innerHTML = JSON.stringify(data.reports, null, 2)

}

async function sendMaintenance(){

const msg = document.getElementById("message").value

await fetch("/api/admin/maintenance",{
method:"POST",
headers:{
"Content-Type":"application/json"
},
body:JSON.stringify({message:msg})
})

alert("Message sent")

}

loadData()

setInterval(loadData,5000)
